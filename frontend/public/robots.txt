# bela-turniri.com — crawler instructions.
# The sitemap is generated dynamically by the backend so it always reflects
# the current set of tournaments and player profiles.

# Default policy for everything not matched below. We block the auth and
# create-tournament routes (no public value indexing them) and the bulk of
# /api/, but Allow the three /api/ subtrees that crawlers legitimately need:
#   /api/preview/    — server-side-rendered OG previews for crawlers
#   /api/resources/  — proxied tournament posters and avatars (og:image targets)
#   /api/sitemap.xml — the sitemap itself
User-agent: *
Disallow: /login
Disallow: /register
Disallow: /tournaments/new
Disallow: /find-pair
Disallow: /api/
Allow: /api/preview/
Allow: /api/resources/
Allow: /api/sitemap.xml

# Social / link-preview crawlers — explicitly allowed everywhere with an
# empty Disallow. Facebook's sharing debugger refuses to scrape unless
# facebookexternalhit is allowlisted; the rest of these are the bots that
# actually drive WhatsApp / Slack / Discord / Twitter previews. WhatsApp
# uses Facebook's crawler under the hood, so allowing facebookexternalhit
# is what fixes WhatsApp link previews too.
